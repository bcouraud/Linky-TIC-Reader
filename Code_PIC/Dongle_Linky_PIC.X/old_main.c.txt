unsigned char ADC_Lecture()
{

}

void delayWithWDT(unsigned int ms) {
    while (ms >= 50) {
        __delay_ms(50);   // delay 50ms at 1MHz
        CLRWDT();         // reset watchdog
        ms -= 50;
    }
    if (ms > 0) {
        __delay_ms(50);
        CLRWDT();
    }
}
